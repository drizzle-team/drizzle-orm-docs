To implement a unique and case-insensitive `email` handling in MSSQL with Drizzle, you can create a persisted computed column with the lowercased `email` and a unique index on it. This way, you can ensure that the `email` is unique regardless of the case.

<CodeTabs items={["schema.ts", "migration.sql"]}>
  <CodeTab>
  ```ts copy {11,14}
  import { sql } from 'drizzle-orm';
  import { int, mssqlTable, uniqueIndex, varchar } from 'drizzle-orm/mssql-core';

  export const users = mssqlTable(
    'users',
    {
      id: int().identity().primaryKey(),
      name: varchar({ length: 255 }).notNull(),
      email: varchar({ length: 255 }).notNull(),
      emailLower: varchar('email_lower', { length: 255 })
        .generatedAlwaysAs(sql`lower(${email})`, { mode: 'persisted' }),
    },
    (table) => [
      uniqueIndex('emailUniqueIndex').on(table.emailLower),
    ],
  );
  ```
  </CodeTab>
  ```sql
  CREATE TABLE [users] (
	[id] int IDENTITY(1, 1),
	[name] varchar(255) NOT NULL,
	[email] varchar(255) NOT NULL,
	[email_lower]  AS (lower(email)) PERSISTED,
	CONSTRAINT [users_pkey] PRIMARY KEY([id])
  );
  --> statement-breakpoint
  CREATE UNIQUE INDEX [emailUniqueIndex] ON [users] ([email_lower]);
  ```
</CodeTabs>

This is how you can select user by `email` with `lower` function:

<Section>
```ts copy {10}
import { eq } from 'drizzle-orm';
import { users } from './schema';

const db = drizzle(...);

const findUserByEmail = async (email: string) => {
  return await db
    .select()
    .from(users)
    .where(eq(sql`lower(${users.email})`, email.toLowerCase()));
};
```

```sql
select * from [users] where lower([users].[email]) = 'john@email.com'
```
