If you want to update your table schema, you can do it in the `schema.ts` file. For example, let's add a new column `phone` to the `users_table`:

```typescript copy filename="src/db/schema.ts" {10}
import { cockroachTable, unique, int4, varchar } from "drizzle-orm/cockroach-core"

export const usersTable = cockroachTable(
  'users',
  {
    id: int4().primaryKey().generatedAlwaysAsIdentity(),
    name: varchar({ length: 255 }).notNull(),
    age: int8({ mode: 'number' }).notNull(),
    email: varchar({ length: 255 }).notNull(),
    phone: varchar(),
  },
  (table) => [
    uniqueIndex('users_email_unique').using('btree', table.email.asc()),
  ],
);
```
